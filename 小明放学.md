# 小明放学

```c++

#include <iostream>
using namespace std;

typedef long long int LL;
const LL N = 100010;

LL r,g,y;

void cal(LL &b, LL &t, LL sum) {
	sum %= (r+g+y);
	while(sum >= t) {
		sum -= t;
		if(b + 1 == 4)
			b = 1;
		else
			b = b + 1;
		if(b == 1)
			t = r;
		else if(b == 2)
			t = g;
		else if(b == 3)
			t = y;
	}
	t -= sum;
	//cout << "  cal " << t << endl; 
}


int main(LL argc, char** argv) {
	cin >> r >> y >> g;
	LL n,b,t;
	cin >> n;
	LL sum = 0;
	for(LL i=0; i<n; i++) {
		cin >> b >> t;
		if(b == 0) {
            sum += t;
            //cout << "    " <<  b <<  "  " << t << endl;
			continue;
		}
		if(b == 2)
			b = 3;
		else if(b == 3)
			b = 2;
		cal(b,t,sum);
		//cout << "    " << b <<  "    " << t << endl;
		if(b == 1) {
			sum += t;
		} else if(b == 3) {
			sum += t + r;
		}
	}
	cout << sum << endl;
	return 0;
}
```

## 代码

```c++

#include <iostream>
#include <bits/stdc++.h>
using namespace std;

typedef long long int LL;
const LL N = 500010;
const LL INF = 0x3f3f3f3f;

struct Edge{
	LL u,v,w;
}; 

LL n,m,s;
LL fa[N];
LL rk[N];

int cmp(Edge a, Edge b){
	return a.w < b.w;
}

void init(){
	memset(fa,0,sizeof(fa));
	memset(rk,0,sizeof(rk));
	for(LL i=1;i<=n;i++){
		fa[i] = i;
	}
}

LL Find(LL x){
	if(x!=fa[x]){
		fa[x] = Find(fa[x]);
	}
	return fa[x];
}

void Union(LL u, LL v){
	LL U = Find(u), V = Find(v);
	if(U == V){
		return ;
	}
	if(rk[U] > rk[V]){
		fa[V] = U; 
	} else {
		fa[U] = V;
		if(rk[U] == rk[V])
			rk[V]++; 
	}
}

int main(int argc, char** argv) {
	Edge t;
	vector<Edge> arr;
	scanf("%lld%lld%lld",&n,&m,&s);
	init();
	for(LL i=0;i<m;i++){
		scanf("%lld%lld%lld",&t.u,&t.v,&t.w);
		arr.push_back(t);
	}
	sort(arr.begin(),arr.end(),cmp);
	LL U,V,cnt = 0,ans=0;
	for(LL i=0; cnt<n-1 && i < arr.size(); i++){
		U = Find(arr[i].u);
		V = Find(arr[i].v);
		//cout << ans << endl;
		if(U != V){
			ans = max(ans, arr[i].w);
			//cout << ans << endl;
			Union(arr[i].u, arr[i].v);
			cnt++;
		}
	}
	printf("%lld\n",ans);
	return 0;
}

```

## 总结

+ 一开始看到这题，以为要用什么高效算法，后来实在想不出，就想暴力，希望枚举所有的生成树，后觉得更不可能，就猜了或许是最小生成树。
+ 此题结点数比较大，用prim可能会超时。用kruskal或许会更好点。
